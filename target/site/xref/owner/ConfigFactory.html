9   package owner;

11  import owner.Config.Sources;

13  import java.io.IOException;
14  import java.io.InputStream;
15  import java.lang.reflect.InvocationHandler;
16  import java.net.MalformedURLException;
17  import java.net.URL;
18  import java.net.URLConnection;
19  import java.util.Properties;
20  
21  import static java.lang.reflect.Proxy.newProxyInstance;
22  import static owner.ConfigURLStreamHandler.CLASSPATH_PROTOCOL;
23  
24  /**
25   * Factory class to instantiate {@link Config} instances. By default a {link Config} sub-interface is associated to a
26   * property having the same package name and class name as the interface itself.
27   * <p/>
28   * Method names are mapped to property names contained in the property files.
29   *
30   * @author Luigi R. Viggiano
31   */
32  public class ConfigFactory {
33      private static final SystemVariablesExpander expander = new SystemVariablesExpander();
34  
35      @SuppressWarnings("unchecked")
36      public static <T extends Config> T create(Class<? extends Config> clazz) {
37          Class<?>[] interfaces = new Class<?>[]{clazz};
38          InvocationHandler handler = new PropertyInvocationHandler(loadPropertiesFor(clazz));
39          return (T) newProxyInstance(clazz.getClassLoader(), interfaces, handler);
40      }
41  
42      static Properties loadPropertiesFor(Class<? extends Config> clazz) {
43          ConfigURLStreamHandler handler = new ConfigURLStreamHandler(clazz.getClassLoader(), expander);
44          try {
45              InputStream stream = getStreamFor(clazz, handler);
46              return load(stream);
47          } catch (IOException e) {
48              throw new RuntimeException(e);
49          }
50      }
51  
52      private static Properties load(InputStream stream) throws IOException {
53          Properties props = new Properties();
54          if (stream != null) {
55              try {
56                  props.load(stream);
57              } finally {
58                  stream.close();
59              }
60          }
61          return props;
62      }
63  
64      static InputStream getStreamFor(Class<? extends Config> clazz, ConfigURLStreamHandler handler) throws IOException {
65          Sources sources = clazz.getAnnotation(Sources.class);
66          if (sources == null)
67              return getDefaultResourceStream(clazz, handler);
68          else
69              return getResourceStreamBySourceAnnotation(sources, handler);
70      }
71  
72      private static InputStream getDefaultResourceStream(Class<? extends Config> clazz,
73                                                          ConfigURLStreamHandler handler) throws IOException {
74          String spec = CLASSPATH_PROTOCOL + ":" + clazz.getName().replace('.', '/') + ".properties";
75          return getInputStream(new URL(null, spec, handler));
76      }
77  
78  
79      private static InputStream getResourceStreamBySourceAnnotation(Sources sources,
80                                                                     ConfigURLStreamHandler handler) throws
81              MalformedURLException {
82  
83          String[] values = sources.value();
84  
85          for (String source : values) {
86              URL url = new URL(null, source, handler);
87              try {
88                  InputStream stream = getInputStream(url);
89                  if (stream != null)
90                      return stream;
91              } catch (IOException ex) {
92                  // ignore: happens when a file specified in the sources is not found or cannot be read.
93              }
94          }
95          return null;
96      }
97  
98      private static InputStream getInputStream(URL url) throws IOException {
99          URLConnection conn = url.openConnection();
100         if (conn == null)
101             return null;
102         return conn.getInputStream();
103     }
